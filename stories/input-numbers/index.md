# Usage

## Input Number
```jsx
  import InputNumber from 'ui/dist/admin/input-number'
  <InputNumber
    autoFocus={...props | optional}
    size={...props | optional}
    min={...props | optional}
    max={...props | optional}
    defaultValue={...props | optional}
    disabled={...props | optional}
    step={...props | optional}
    value={...props}
    onChange={this.onInputNumberChange}
  />
```
