# Usage

## Alert

```jsx
<Alert message='I am an alert' type='success />
```

```jsx
  import Alert from 'ui/dist/admin/alert'
  <Alert
      message={...props}
      description={...props | optional}
      type={...props | optional}
      closable={...props | optional}
      closeText={...props | optional}
      showIcon={...props | optional}
      icon={...props | optional}
      onClose={this.onClose}
      afterClose={this.afterClose}
    />
```
